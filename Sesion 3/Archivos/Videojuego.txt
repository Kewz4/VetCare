public class Videojuego
    {
        // ATRIBUTOS (propiedades del videojuego)
        private string titulo;
        private int anioLanzamiento;
        private double calificacion;  // De 0 a 10

        // Constructor - Inicializa el videojuego
        public Videojuego(string titulo, int anio, double calificacion)
        {
            this.titulo = titulo;
            this.anioLanzamiento = anio;
            this.calificacion = calificacion;
        }

        // MÉTODOS EXISTENTES
        public void MostrarInfo()
        {
            Console.WriteLine($"\n🎮 --- {titulo} ---");
            Console.WriteLine($"Año de lanzamiento: {anioLanzamiento}");
            Console.WriteLine($"Calificación: {calificacion}/10 ⭐");
        }

        public bool EsJuegoModerno()
        {
            return anioLanzamiento >= 2020;
        }

        public void MostrarEstado()
        {
            if (calificacion >= 8)
                Console.WriteLine("✅ ¡Excelente juego!");
            else if (calificacion >= 6)
                Console.WriteLine("👍 Juego bueno");
            else
                Console.WriteLine("⚠️ Juego promedio");
        }
